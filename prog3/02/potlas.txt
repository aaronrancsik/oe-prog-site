{
 var q2 = people
     .Where(x => x.Dept == dept)

     .Take(pagesize)

var q3 = from person in people
      let minlen = people.Min(x => x.Name.Length)
      let maxlen = people.Max(x => x.Name.Length)
      where person.Name.Length == minlen ||
      person.Name.Length == maxlen
      select new { person.Name, person.Name.Length };
q3.ToConsole("Q3");
